// Sp food items data const foodItemsData = [ { id: 1, name: "Pizza", price: 8.99 }, { id: 2, name: "Burger", price: 8.99 }, { id: 3, name: "shushi", price: 8.99 }, { id: 4, name: "salad", price: 8.99 }, { id: 5, name: "desserts", price: 9.99 }, { id: 6, name: "chipo mwitu", price: 12.50 }, { id: 7, name: "chappo", price: 2.50 }, ]; // Display food items on the page function displayFoodItems() { const foodItemsDiv = document.getElementById("foodItems"); foodItemsData.forEach(item => { const foodItem = document.createElement("div"); foodItem.classList.add("food-item"); foodItem.innerHTML = ` ${item.name} $${item.price} `; foodItemsDiv.appendChild(foodItem); }); } // Add item to the cart function addToCart(itemId) { const selectedItem = foodItemsData.find(item => item.id === itemId); const cartItemsUl = document.getElementById("cartItems"); const cartItem = document.createElement("li"); cartItem.innerHTML = ` ${selectedItem.name} $${selectedItem.price} `; cartItemsUl.appendChild(cartItem); } // Initialize the app function init() { displayFoodItems(); const checkoutBtn = document.getElementById("checkoutBtn"); checkoutBtn.addEventListener("click", () => { alert("Thank you for your order welcome back again@Sp!"); }); } // Call init when the DOM is ready document.addEventListener("DOMContentLoaded", init);
